$("#checkall").click(function() {
    $("input:checkbox").each(function() {
        $(this).attr('checked','')

    $(".layui-form-checkbox").each(function() {
        $(this).addClass('layui-form-checked');

    $("input:checkbox").each(function() {
        $(this).removeAttr('checked');
    });
    $(".layui-form-checkbox").each(function() {
        $(this).removeClass('layui-form-checked');
